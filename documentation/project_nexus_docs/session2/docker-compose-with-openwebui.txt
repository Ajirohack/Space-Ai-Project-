version: '3.8'

services:
  # Frontend service
  nexus-client:
    build:
      context: ./client
    ports:
      - "3000:3000"
    depends_on:
      - nexus-server
    networks:
      - nexus-network
    restart: unless-stopped

  # Backend service
  nexus-server:
    build:
      context: ./server
    ports:
      - "5000:5000"
    volumes:
      - ./server/uploads:/app/uploads
      - ./server/data:/app/data
    environment:
      - NODE_ENV=production
      - PORT=5000
      - LLM_PROVIDER=${LLM_PROVIDER:-openwebui}
      - LLM_API_KEY=${LLM_API_KEY:-not-required}
      - OPENWEBUI_API_URL=http://openwebui:1234/v1
    networks:
      - nexus-network
    restart: unless-stopped
    depends_on:
      - openwebui

  # Open Web UI service
  openwebui:
    image: ghcr.io/open-webui/open-webui:latest
    container_name: openwebui
    restart: unless-stopped
    ports:
      - "8080:8080"  # Web UI
      - "1234:1234"  # API Port
    volumes:
      - ./open-webui-data:/app/backend/data
    networks:
      - nexus-network
    environment:
      - OLLAMA_API_BASE_URL=http://ollama:11434/api
      - PORT=8080
      - OPENAI_API_KEY=${OPENAI_API_KEY:-not-required}
    depends_on:
      - ollama

  # Ollama service for running local models
  ollama:
    image: ollama/ollama:latest
    container_name: ollama
    restart: unless-stopped
    ports:
      - "11434:11434"
    volumes:
      - ./ollama-data:/root/.ollama
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: all
              capabilities: [gpu]
    networks:
      - nexus-network

  # Optional MongoDB for more robust memory storage
  mongodb:
    image: mongo:latest
    container_name: nexus-mongodb
    restart: unless-stopped
    ports:
      - "27017:27017"
    volumes:
      - ./mongodb-data:/data/db
    networks:
      - nexus-network
    environment:
      - MONGO_INITDB_ROOT_USERNAME=${MONGO_USER:-nexus}
      - MONGO_INITDB_ROOT_PASSWORD=${MONGO_PASSWORD:-nexuspassword}

  # Optional Redis for caching and working memory
  redis:
    image: redis:alpine
    container_name: nexus-redis
    restart: unless-stopped
    ports:
      - "6379:6379"
    volumes:
      - ./redis-data:/data
    networks:
      - nexus-network

networks:
  nexus-network:
    driver: bridge

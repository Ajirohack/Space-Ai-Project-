import React from 'react';

function Message({ message, isNexus }) {
  const renderAttachment = (attachment) => {
    switch (attachment.type) {
      case 'image':
        return (
          <div className="attachment image-attachment">
            <img src={attachment.url} alt={attachment.name} />
          </div>
        );
      case 'document':
        return (
          <div className="attachment document-attachment">
            <a href={attachment.url} target="_blank" rel="noopener noreferrer">
              <div className="document-icon">
                ðŸ“„
              </div>
              <div>
                <div>{attachment.name}</div>
                <div className="attachment-size">{attachment.size || 'Document'}</div>
              </div>
            </a>
          </div>
        );
      case 'audio':
        return (
          <div className="attachment audio-attachment">
            <audio controls src={attachment.url}></audio>
            <span>Voice message</span>
          </div>
        );
      default:
        return null;
    }
  };

  return (
    <div className={`message ${isNexus ? 'nexus-message' : 'user-message'}`}>
      {isNexus && (
        <div className="nexus-avatar">
          <div className="nexus-avatar-inner">
            <div className="nexus-avatar-pulse"></div>
          </div>
        </div>
      )}
      <div className="message-content">
        <div className="message-text">{message.text}</div>
        {message.attachments && message.attachments.map((attachment, index) => (
          <div key={index}>
            {renderAttachment(attachment)}
          </div>
        ))}
        <div className="message-time">{message.timestamp}</div>
      </div>
    </div>
  );
}

export default Message;
